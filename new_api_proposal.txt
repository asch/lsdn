#ifndef _LSDN_H
#define _LSDN_H

struct lsdn_context;

lsdn_context_create()
/* Realize all changes since the last commit */
lsdn_commit(ctx)

/* Description of a virtual network */
struct lsdn_net;
lsdn_net_create_vlan(context, vlan_id)
lsdn_net_create_vxlan(context, vxlan_id)

/* Connection to a network at a physical level */
struct lsdn_phys;
lsdn_phys_create(context)
/* needed by vxlan */
lsdn_phys_set_ip(phys, ip)
/* needed if the phys is local */
lsdn_phys_set_iface(phys, iface_name)
/* connect the phys locally to this computer*/
lsdn_phys_claim_local(phys)
lsdn_phys_attach(phys, net)


/* Connection to a network at a virtual level (realized throught a phys) */
struct lsdn_virt;
lsdn_virt_create(net)
lsdn_virt_set_ip_range(virt, ip_range)
lsdn_virt_set_mac_range(virt, mac_range)
/* will include migration */
lsdn_virt_connect(virt, phys, interface)

lsdn_cfg_firewall/qos(virt)

#endif
